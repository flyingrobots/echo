\RequirePackage{tikz-cd}
\tikzcdset{row sep=large, column sep=large}
% \DPO{K}{L}{R}{D}{G}{H} renders two adjacent pushout squares:
%   L \leftarrow K \rightarrow R
%   \downarrow    \downarrow    \downarrow
%   G \leftarrow D \rightarrow H
% Used to illustrate double-pushout rewrites; parameters are (K,L,R,D,G,H).
\newcommand{\DPO}[6]{%
\begin{center}
\begin{tikzcd}[ampersand replacement=\&]
#1 \arrow[r, hook] \arrow[d, hook] \& #2 \arrow[d, hook] \& \qquad
#1 \arrow[r, hook] \arrow[d, hook] \& #3 \arrow[d, hook] \\
#4 \arrow[r, hook] \& #5 \& \qquad #4 \arrow[r, hook] \& #6
\end{tikzcd}
\end{center}%
}
